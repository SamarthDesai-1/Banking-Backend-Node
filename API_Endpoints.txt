For Registration {

  API : {
    Signup : http://localhost:5000/test/api/users/signup
    Verify via OTP : http://localhost:5000/test/api/users/verify
  }
}

For Login {

  API : {
    Login : https://localhost:5000/test/api/users/login
    Profile or Home page : https://localhost:5000/test/api/users/profile
  }
}

For Forget Password {

  API : {
    Forget-password : https://localhost:5000/test/api/users/forget-password
    Reset-password : localhost:5000/test/api/users/reset-password?token= { token generated by random string }
  }
}
